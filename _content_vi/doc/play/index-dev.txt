package main

import (
	"fmt"
)

// Hàm index trả về chỉ số của lần xuất hiện đầu tiên của v trong s,
// hoặc -1 nếu không tìm thấy.
func index[E comparable](s []E, v E) int {
	for i, vs := range s {
		if v == vs {
			return i
		}
	}
	return -1
}

func main() {
	s := []int{1, 3, 5, 2, 4}
	fmt.Println(index(s, 3))
	fmt.Println(index(s, 6))
}
